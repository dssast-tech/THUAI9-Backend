# THUAI8 后端与通信组分工方案

## 一、现状分析

### 当前架构问题

**后端代码结构**：
```
Program.cs
  ↓ 启动 gRPC 服务器
  ↓ 启动游戏主循环 (Env.run())

Env.cs (游戏主循环)
  ↓ 每回合调用 BroadcastGameState()
  ↓ 等待 gRPC 客户端发送操作
  ↓ 执行操作
  ↓ 循环
```

**问题**：
- ❌ 游戏主循环在 Server 内部
- ❌ 使用 gRPC 通信
- ❌ 需要两个独立的 client 连接

**Saiblo 要求**：
- ✅ 游戏主循环在 Game Logic 容器内
- ✅ 使用 stdin/stdout 通信
- ✅ AI 由 Saiblo Worker 管理

## 二、重构方案

### 方案选择

**推荐：方案 A - 轻量级重构**

**核心思路**：
- 保留后端的游戏逻辑代码（Env.cs, Piece.cs, Board.cs 等）
- 移除 gRPC 相关代码
- 添加 JSON 序列化/反序列化接口
- 游戏主循环移到 Python 适配器中

**优点**：
- 后端改动最小（主要是删除 gRPC，添加 JSON 接口）
- 通信组可以用 Python 调用 C# 库
- 架构清晰，易于维护

## 三、详细分工

import clr
import sys
import json

# 加载 C# DLL
sys.path.append('./server')
clr.AddReference('server')

from Server import GameEngine

class GameEngineWrapper:
    """C# GameEngine 的 Python 包装器"""

    def __init__(self):
        self.engine = GameEngine()

    def initialize(self, config: dict):
        """初始化游戏"""
        config_json = json.dumps(config)
        self.engine.Initialize(config_json)

    def set_player_pieces(self, player_id: int, pieces: list) -> bool:
        """配置玩家棋子"""
        pieces_json = json.dumps(pieces)
        return self.engine.SetPlayerPieces(player_id, pieces_json)

    def get_state(self) -> dict:
        """获取游戏状态"""
        state_json = self.engine.GetStateJson()
        return json.loads(state_json)

    def execute_action(self, player_id: int, action: dict) -> bool:
        """执行玩家操作"""
        action_json = json.dumps(action)
        return self.engine.ExecuteAction(player_id, action_json)

    def next_turn(self):
        """推进到下一回合"""
        self.engine.NextTurn()

    def is_game_over(self) -> bool:
        """检查游戏是否结束"""
        return self.engine.IsGameOver()

    def get_winner(self) -> int:
        """获取胜者"""
        return self.engine.GetWinner()

    def get_replay(self) -> dict:
        """获取回放数据"""
        replay_json = self.engine.GetReplayJson()
        return json.loads(replay_json)

#### 任务 3.2：重写 Game Host 主程序（1-2天）

**通信组负责**：

```python
# main.py
import sys
from saiblo_protocol import SaibloProtocol, EndState
from game_engine_wrapper import GameEngineWrapper

class GameHost:
    """Game Host 主类"""

    def __init__(self):
        self.engine = GameEngineWrapper()
        self.replay_file = None
        self.current_round = 0
        self.player_states = {0: EndState.OK, 1: EndState.OK}

    def initialize(self):
        """初始化阶段"""
        print("[INFO] 初始化阶段", file=sys.stderr)

        # 1. 读取 Saiblo 初始化消息
        init_msg = SaibloProtocol.read_message()
        init_info = SaibloProtocol.parse_init_message(init_msg)

        # 2. 打开回放文件
        self.replay_file = open(init_info["replay"], 'w')

        # 3. 初始化游戏引擎
        self.engine.initialize(init_info["config"])

        # 4. 配置玩家棋子（使用默认配置）
        for player_id in [0, 1]:
            pieces = self.get_default_pieces(player_id)
            self.engine.set_player_pieces(player_id, pieces)

        # 5. 发送回合配置给 Saiblo
        SaibloProtocol.send_round_config(time=60, length=4096)

        print("[INFO] 初始化完成", file=sys.stderr)

    def game_loop(self):
        """游戏主循环"""
        print("[INFO] 游戏循环开始", file=sys.stderr)

        while not self.engine.is_game_over():
            self.current_round += 1

            # 1. 获取游戏状态
            state = self.engine.get_state()
            current_player = state["current_player"]

            print(f"[INFO] 回合 {self.current_round}, 玩家 {current_player}", file=sys.stderr)

            # 2. 写入回放
            state_json = json.dumps(state)
            self.replay_file.write(state_json + '\n')
            self.replay_file.flush()

            # 3. 发送游戏状态给 Saiblo
            SaibloProtocol.send_round_info(
                state=self.current_round,
                listen=[current_player],
                players=[0, 1],
                content=[state_json, state_json]
            )

            # 4. 接收 AI 操作
            ai_msg = SaibloProtocol.read_message()
            ai_info = SaibloProtocol.parse_ai_message(ai_msg)

            # 5. 处理异常
            if ai_info["is_error"]:
                self.handle_ai_error(ai_info)
                break

            # 6. 执行操作
            action = json.loads(ai_info["content"])
            self.engine.execute_action(current_player, action)

            # 7. 推进回合
            self.engine.next_turn()

        # 游戏结束
        self.handle_game_end()

        print("[INFO] 游戏循环结束", file=sys.stderr)

    def handle_game_end(self):
        """处理游戏结束"""
        winner = self.engine.get_winner()

        if winner == -1:
            end_info = {"0": 0, "1": 0}
        else:
            end_info = {str(winner): 1, str(1 - winner): 0}

        end_state = [self.player_states[0], self.player_states[1]]

        SaibloProtocol.send_game_end(end_info, end_state)

    def cleanup(self):
        """清理资源"""
        if self.replay_file:
            self.replay_file.close()

    def run(self):
        """运行 Game Host"""
        try:
            self.initialize()
            self.game_loop()
        except Exception as e:
            print(f"[ERROR] {e}", file=sys.stderr)
            import traceback
            traceback.print_exc()
        finally:
            self.cleanup()

def main():
    game_host = GameHost()
    game_host.run()

if __name__ == "__main__":
    main()
```

def send_message(proc, data):
    content = json.dumps(data).encode('utf-8')
    length = struct.pack('>I', len(content))
    proc.stdin.write(length + content)

    length_bytes = proc.stdout.read(4)
